const calculateConsistency = (entries: ProgressEntry[]): number => {
  const practiceCount = entries.filter(e => e.netHours > 0).length;
  return (practiceCount / entries.length) * 100;
};

// Improvement Tracking
const calculateImprovement = (entries: ProgressEntry[]): number => {
  const recent = entries.slice(-3);
  const older = entries.slice(-6, -3);
  const recentAvg = recent.reduce((sum, e) => sum + e.selfRating, 0) / recent.length;
  const olderAvg = older.reduce((sum, e) => sum + e.selfRating, 0) / older.length;
  return ((recentAvg - olderAvg) / olderAvg) * 100;
};
